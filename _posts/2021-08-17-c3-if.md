# 증감 연산자
```c
int main11() {
	int num1 = 1;
	int num2 = 2;

	num1++; // num1 = num1 + 1;
	num2--; // num2 = num2 - 1;

	printf("%d\n", num2);
	printf("%d\n", num1); 


	//실수형

	float num3 = 2.1f;
	float num4 = 2.1f;

	num3++;
	num4--;

	printf("%f %f\n", num3, num4); //3.10000000, 1.10000000
	
	//문자 자료형
	char c1 = 'b';
	char c2 = 'b';

	c1++;
	c2--;

	printf("%c %c", c1, c2); // c a

	return 0;
}

```


# 곱셈 나눗셈
```c
int main12() {
	int num1;
	int num2;

	num1 = 2 * 3;
	num2 =  7 / 2;

	printf("%d", num1); //6
	printf("%d", num2); //3 정수끼리 나눗셈하면 정수가 나옴 0으로 나누면 컴파일 에러 발생.

	num1 *= 3;
	num2 /= 2;


	float num3;
	float num4;

	num3 = 2.73f * 3.81f;
	num4 = 7.0f / 2.0f;

	printf("%f\n", num3); //10.401299 원래는 10.4013 실수계산 오차가 발생함 동일하게 0으로 나누면 컴파일 에러
	printf("%f\n", num4); // 3.500000

}
```

# 나머지 연산하기
```c
int main13() {

	printf("%d\n", 1 % 3); //1

	printf("%d\n", 4 % 3); //1
```

# 자료형의 확장과 축소 알아보기
```c
int main14() {
	 //자료형의 확장
	int  num1 = 11;
	float num2 = 4.4f;

	printf("%f\n", num1 + num2); //15.400000
	printf("%f\n", num1 - num2); // 6.600000
	printf("%f\n", num1 * num2); // 48.400002
	printf("%f\n", num1 / num2); // 2.500000

	//실수랑 정수를 계산하면 범위가넓은 실수로 자동 변환이 된다.(표현범위가 넓다) =>암시적 형 변환

	long long num3 = 123456789000;
	int num4 = 10;

	printf("%lld\n", num3 + num4); //123456789010
	printf("%lld\n", num3 - num4); //123456788990
	printf("%lld\n", num3 * num4); //1234567890000
	printf("%lld\n", num3 / num4); //12345678900

	//자료형의 축소

	float num5 = 11.0f;
	float num6 = 5.0f;

	int num7 = num5 / num6; // 2 자료형의 크기가 작은쪽, 표현 범위가 좁은 쪽을 형 축소 라고함. (값의 손실 발생)

	printf("%d\n", num7);
	// 문자 => 정수 => 실수 로 갈수록 형 확장 반대로는 형 축소
	return 0;

}
```


# if 조건문으로 특정 조건일 때 코드 실행하기
```c
int main15() {
	int num1 = 10;
	if (num1 == 10) //끝에 세미콜론 쓰는 거 주의!
	{
		printf("10 입니다.\n");
	}

	if (num1 == 10)
		printf("10입니다"); //실행코드가 한줄이면 중괄호 생략 가능


	//실수와 정수 비교 하기
	float num2 = 0.1f;
	char c1 = 'a';

	if (num2 == 0.1f)
		printf("0.1 입니다.\n");

	if (c1 == 'a')
		printf("a입니다.\n");

	if (c1 == 97) //아스키 코드
		printf("a입니다");


	//사용자 값 받아서 비교하기
	// #define _CRT_SECURE_NO_WARNINGS 

	int num4;

	scanf("%d", &num4);

	if (num4 == 10)
		printf("10입니다.");

	if (num4 == 20)
		printf("20입니다.");
	return 0;
}

```
