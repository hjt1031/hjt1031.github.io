# goto로 프로그램의 흐름을 원하는 대로 바꾸기
```c
int main28() {
	//define _CRT_SECURES_NO_WARNINGS

	int num1;

	scnaf("%d", &num1);
	if (num1 == 1)
		goto ONE;
	else if (num1 == 2)
		goto TWO;
	else
		goto EXIT;

ONE:
	printf("1입니다.");
	goto EXIT;

TWO:
	printf("2입니다.");
	goto EXIT;

EXIT:
	return 0;


	//중첩 루프 빠져나오기
	//#include <stdbool.h>

	int num1 = 0;

	bool exitOuterLoop = false;
	for (int i = 0; i < 10; i++) {
		for (int j = 0; j < 10; j++) {
			if (num1 == 20) {
				exitOuterLoop = true;
				break;
			}

			num1++;
		}
		if (exitOuterLoop == true)
			break;
	}
	printf("%d\n", num1);
	//goto 사용

	for (int i = 0; i < 10; i++) {
		for (int j = 0; j < 10; j++) {
			if (num1 == 20) {
				goto EXIT;
			}

			num1++;
		}
	}

EXIT:
	printf("%d\n", num1);

	//goto와 에러 처리 패턴

	int gender;
	int age;
	bool isOwner;

	scanf("%d %d %d", &gender, &age ,&isOwner);

	printf("안녕하세요\n");
	printf("문을 연다.\n");

	if (gender == 2)
	{
		goto EXIT;
	}

	if (age < 30) {
		goto EXIT;
	}

	if (isOwner == false) {
		goto EXIT;
	}
EXIT:
	printf("안녕하세요.\n");
	printf("문을 닫는다.");

	return 0;
}
```
