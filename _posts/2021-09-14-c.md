# 입력 값을 배열 형태의 문자열에 저장하기
```c
int main32() {
	//define _CRT_SECURE_NO_WARNINGS

	char s1[10]; //배열의 크기 -1 로 입력해야함 맨마지막은 null

	printf("문자열을 입력하세요 : ");
	scanf("%s", s1);

	printf("%s", s1);

	//입력 값을 문자열 포인터에 저장하기

	char* s1 = malloc(sizeof(char) * 10);

	printf("문자열을 입력하세요 : ");
	scanf("%s", s1);

	printf("%s", s1);

	free(s1);

	//문자열을 여러개 받기

	char s1[10];
	char s2[10];

	printf("문자열 두개를 입력하세요 : ");
	scanf("%s %s", s1, s2);

	printf("%s", s1);
	printf("%s", s2);


	return 0;
}
```

# 문자열 길이 구하기
```c
int main33() {
	//#include <string.h>

	char* s1 = "Hello";
	char s2[10] = "hello";

	printf("%d\n", strlen(s1)); // 5
	printf("%d\n", strlen(s2)); // 5
	

	//문자열 비교하기
	char s1[10] = "Hello";
	char* s2 = "Hello";

	int ret = strcmp(s1, s2); //비교 문자열 넣는 것 정수로 반환함(대소문자 구분함)

	printf("%d\n", ret); // 0

	printf("%d\n", strcmp("aaa","aaa")); // 0
	printf("%d\n", strcmp("aab","aaa")); // 1 앞에가 큰경우
	printf("%d\n", strcmp("aab","aac")); // -1 뒤에가 큰경우


	char s1[20];
	char s2[20];

	printf("문자열 두 개를 입력하세요 : ");
	scanf("%s %s", s1, s2);

	int ret = strcmp(s1, s2);

	switch (ret)
	{
	case 0 :
		printf("두 문자열이 같음");
		break;
	case 1:
		printf("%s보다 %s가 큼 ", s2, s1);
		break;
	case -1:
		printf("%s보다 %s가 작음", s2, s1);
		break;
	}
	return 0;
}
```

# 문자열 복사하기
```c
int main33() {

	char s1[10] = "Hello";
	char s2[10];

	strcpy(s2, s1);

	printf("%s", s2); //Hello


	//문자열 붙이기
	char s1[10] = "world";
	char s2[20] = "Hello";

	strcar(s2, s1);
	
	printf("%s\n", s2); //Hello world


	char *s1 = "world";
	char* s2 = malloc(sizeof(char) * 20);

	strcar(s2, "Hello");

	strcat(s2, s1);

	printf("%s\n", s2); //Hello world

	free(s2);


	//배열형태의 문자열을 문자열 포인터에 복사하기
	char s1[10] = "Hello";
	char* s2 = malloc(sizeof(char) * 10);

	strcpy(s2, s1);

	printf("%s\n", s2); //Hello
	free(s2);

	//배열 형태의 문자열을 문자열 포인터에 붙이기
	char s1[10] = "world";
	char* s2 = malloc(sizeof(char) * 20);

	strcpy(s2, "Hello");

	strcat(s2, s1);

	printf("%s\n", s2); //Helloworld

	free(s2);

	return 0;
}
```
