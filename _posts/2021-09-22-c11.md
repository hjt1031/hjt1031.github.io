# 구조체 사용하기 (struct로 정의)
```c
//관련 정보를 하나의 의미로 묶을 때 사용  한다.
//기본 자료형 (char, int, float)등을 조합하여 만든 자료형을 파생형 이라고 한다.
struct Person{
	char name[20];
	int age;
	char address[100];
};

//typedef 로 키워드 생략 하기
typedef struct _Person{
	char name[20];
	int age;
	char address[100];
} Person

//익명 구조체 사용하기
typedef struct {
	char name[20];
	int age;
	char address[100];0
} Person;
int main() {
	struct Person p1;
	
	strcpy(p1.name, "홍길동");
	p1.age = 30;
	strcpy(p1.address,"서울시 용산구 한남동");
	
	printf("이름",p1.name);
	printf("나이",p1.age);
	printf("주소",p1.address);
	
	//typedef 로 키워드 생략 하기
	Person p1;
	
	strcpy(p1.name, "홍길동");
	p1.age = 30;
	strcpy(p1.address,"서울시 용산구 한남동");
	
	printf("이름",p1.name);
	printf("나이",p1.age);
	printf("주소",p1.address);
	
	//익명 구조체 사용하기
	Person p1;
	
	strcpy(p1.name, "홍길동");
	p1.age = 30;
	strcpy(p1.address,"서울시 용산구 한남동");
	
	printf("이름",p1.name);
	printf("나이",p1.age);
	printf("주소",p1.address);
	return 0;
}

```


# 구조체 포인터로 선언하고 메모리 할당하기
```c
struct Person{
	char name[20];
	int age;
	char address[100];
};

//구조체 별칭으로 포인터를 선언하고 메모리 할당하기
typedef struct _Person{
	char name[20];
	int age;
	char address[100];
} Person;

//구조체 포인터에 구조체 변수의 주소 할당하기
struct _Person{
	char name[20];
	int age;
	char address[100];
};
int main() {
	struct Person *p1 = malloc(sizeof(struct Person));
	
	strcpy(p1->name, "홍길동");
	p1->age = 30;
	strcpy(p1->address, "서울시 용산구 한남동");
	
	printf("이름 %s",p1->name);
	printf("나이 %d",p1->age);
	printf("주소 %s",p1->address);
	
	free(p1);
	
	//구조체 별칭으로 포인터를 선언하고 메모리 할당하기
	
	Person *p1 = malloc(sizeof(Person));
	
	strcpy(p1->name, "홍길동");
	p1->age = 30;
	strcpy(p1->address, "서울시 용산구 한남동");
	
	printf("이름 %s",p1->name);
	printf("나이 %d",p1->age);
	printf("주소 %s",p1->address);
	
	free(p1);
	
	//구조체 포인터에 구조체 변수의 주소 할당하기
	struct Person p1;
	struct Person *ptr;
	
	ptr = &p1;
	
	ptr->age = 30;
	
	printf("나이 %d", p1.age);
	printf("나이 %d", ptr->age);
	return 0;
}
```

# 두 점 사이의 거리 구하기
```c
struct Point2D {
	int x;
	int y;
};
int main() {
	struct Point2D p1;
	struct Point2D p2;
	
	p1.x = 30;
	p1.y = 20;
	
	p2.x = 60;
	p2.y = 50;
	
	printf("%d%d", p1.x, p1.y); // 30 20
	printf("%d%d", p2.x, p2.y); // 60 50
	
	int a = p2.x - p1.x; //선 a의 길이
	int b = p2.y - p1.y; //선 b의 길이

	//sqrt (루트) #include<math.h>
	//doublde c = sqrt((a*a) + (b*b)); //거듭제곱
	doublde c = sqrt(pow(a,2) + pow(b,2)); //거듭제곱
	
	printf("%f\n",c); //42.426407
	return 0;
}

//구조체 멤버 정렬 사용하기
struct PacketHeader{
	char flags;
	int seq;
}
//구조체 정렬 크기 조절하기
#pragma pack(push,1) //1바이트 크기로 정렬 함.
struct PacketHeader{
	char flags;
	int seq;
}
#pragma pack(pop,1)
int main() {
	
	struct PacketHeader header;
	
	printf("%d\n", sizeof(header.flags)); //1
	printf("%d\n", sizeof(header.seq)); //4
	printf("%d\n", sizeof(header)); //8
	printf("%d\n", sizeof(struct PacketHeader));//8
	
	// include <stddef.h> => 메모리 위치 확인 offsetof 함수로
	printf("%d\n", offsetof(struct PacketHeader, flags));//0
	printf("%d\n", offsetof(struct PacketHeader, seq));//4
	
	
	//구조체 정렬 크기 조절하기
	struct PacketHeader header;
	
	printf("%d\n", sizeof(header.flags)); // 1
	printf("%d\n", sizeof(header.seq)); // 4
	printf("%d\n", sizeof(header)); // 5
	// include <stddef.h>
	printf("%d\n", offsetof(struct PacketHeader, flags)); // 0
	printf("%d\n", offsetof(struct PacketHeader, seq)); // 1
	return 0;
};
```
# 구조체와 메모리를 간단하게 0으로 설정하기
```c
struct Point2D {
	int x;
	int y;
}
int main() {
	struct Point2D p1;
	
	memset(&p1, 0, sizeof(struct Point2D));
	
	printf("%d %d\n",p1.x, p1.y); // 0 0
	
	//다른방법
	struct Point2D *p1 = malloc(sizeof(struct Point2D));
	
	memset(p1, 0, sizeof(struct Point2D));
	
	printf("%d %d\n",p1->x, p1->y); // 0
	
	free(p1);
	
	//구조체와 메모리 복사하기
	struct Point2D p1;
	struct Point2D p2;
	
	p1.x = 10;
	p1.y = 20;
	
	memcpy(&p2, &p1, sizeof(struct Point2D));
	
	printf("%d %d\n", p2.x, p2.y);
	
	//메모리 복사 다른방법 1
	struct Point2D *p1 = malloc(sizeof(struct Point2D));
	struct Point2D *p2 = malloc(sizeof(struct Point2D));
	
	p1->x = 10;
	p1->y = 20;
	
	memcpy(p2, p1, sizeof(struct Point2D));
	
	printf("%d %d\n", p2->x, p2->y);
	
	free(p2);
	free(p1);
	
	//메모리 복사 다른방법 2
	struct Point2D *p1 = malloc(sizeof(struct Point2D));
	struct Point2D p2;
	
	memcpy(&p2, p1, sizeof(struct Point2D));
	
	free(p1);
}
```
