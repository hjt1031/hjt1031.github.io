# 구조체 사용하기 (struct로 정의)
```c
//관련 정보를 하나의 의미로 묶을 때 사용  한다.

struct Person{
	char name[20];
```

# 구조체 포인터로 선언하고 메모리 할당하기
```c
struct Person{
	char name[20];
	int age;
	char address[100];
};

//구조체 별칭으로 포인터를 선언하고 메모리 할당하기
typedef struct _Person{
	char name[20];
	int age;
	char address[100];
} Person;

//구조체 포인터에 구조체 변수의 주소 할당하기
struct _Person{
	char name[20];
	int age;
	char address[100];
};
int main() {
	struct Person *p1 = malloc(sizeof(struct Person));
	
	strcpy(p1->name, "홍길동");
	p1->age = 30;
	strcpy(p1->address, "서울시 용산구 한남동");
	
	printf("이름 %s",p1->name);
	printf("나이 %d",p1->age);
	printf("주소 %s",p1->address);
	
	free(p1);
	
	//구조체 별칭으로 포인터를 선언하고 메모리 할당하기
	
	Person *p1 = malloc(sizeof(Person));
	
	strcpy(p1->name, "홍길동");
	p1->age = 30;
	strcpy(p1->address, "서울시 용산구 한남동");
	
	printf("이름 %s",p1->name);
	printf("나이 %d",p1->age);
	printf("주소 %s",p1->address);
	
	free(p1);
	
	//구조체 포인터에 구조체 변수의 주소 할당하기
	struct Person p1;
	struct Person *ptr;
	
	ptr = &p1;
	
	ptr->age = 30;
	
	printf("나이 %d", p1.age);
	printf("나이 %d", ptr->age);
	return 0;
}
```

# 두 점 사이의 거리 구하기

```c
struct Point2D {
	int x;
	int y;
};
int main() {
	struct Point2D p1;
	struct Point2D p2;
	
	p1.x = 30;
	p1.y = 20;
	
	p2.x = 60;
	p2.y = 50;
	
	printf("%d%d", p1.x, p1.y); // 30 20
	printf("%d%d", p2.x, p2.y); // 60 50
	
	int a = p2.x - p1.x; //선 a의 길이
	int b = p2.y - p1.y; //선 b의 길이

	//sqrt (루트) #include<math.h>
	//doublde c = sqrt((a*a) + (b*b)); //거듭제곱
	doublde c = sqrt(pow(a,2) + pow(b,2)); //거듭제곱
	
	printf("%f\n",c); //42.426407
	return 0;
}

//구조체 멤버 정렬 사용하기
struct PacketHeader{
	char flags;
	int seq;
}
//구조체 정렬 크기 조절하기
#pragma pack(push,1) //1바이트 크기로 정렬 함.
struct PacketHeader{
	char flags;
	int seq;
}
#pragma pack(pop,1)
int main() {
	
	struct PacketHeader header;
	
	printf("%d\n", sizeof(header.flags)); //1
	printf("%d\n", sizeof(header.seq)); //4
	printf("%d\n", sizeof(header)); //8
	printf("%d\n", sizeof(struct PacketHeader));//8
	
	// include <stddef.h> => 메모리 위치 확인 offsetof 함수로
	printf("%d\n", offsetof(struct PacketHeader, flags));//0
	printf("%d\n", offsetof(struct PacketHeader, seq));//4
	
	
	//구조체 정렬 크기 조절하기
	struct PacketHeader header;
	
	printf("%d\n", sizeof(header.flags)); // 1
	printf("%d\n", sizeof(header.seq)); // 4
	printf("%d\n", sizeof(header)); // 5
	// include <stddef.h>
	printf("%d\n", offsetof(struct PacketHeader, flags)); // 0
	printf("%d\n", offsetof(struct PacketHeader, seq)); // 1
	return 0;
};
```

# 구조체와 메모리를 간단하게 0으로 설정하기
```c
struct Point2D {
	int x;
	int y;
}
int main() {
	struct Point2D p1;
	
	memset(&p1, 0, sizeof(struct Point2D));
	
	printf("%d %d\n",p1.x, p1.y); // 0 0
	
	//다른방법
	struct Point2D *p1 = malloc(sizeof(struct Point2D));
	
	memset(p1, 0, sizeof(struct Point2D));
	
	printf("%d %d\n",p1->x, p1->y); // 0
	
	free(p1);
	
	//구조체와 메모리 복사하기
	struct Point2D p1;
	struct Point2D p2;
	
	p1.x = 10;
	p1.y = 20;
	
	memcpy(&p2, &p1, sizeof(struct Point2D));
	
	printf("%d %d\n", p2.x, p2.y);
	
	//메모리 복사 다른방법 1
	struct Point2D *p1 = malloc(sizeof(struct Point2D));
	struct Point2D *p2 = malloc(sizeof(struct Point2D));
	
	p1->x = 10;
	p1->y = 20;
	
	memcpy(p2, p1, sizeof(struct Point2D));
	
	printf("%d %d\n", p2->x, p2->y);
	
	free(p2);
	free(p1);
	
	//메모리 복사 다른방법 2
	struct Point2D *p1 = malloc(sizeof(struct Point2D));
	struct Point2D p2;
	
	memcpy(&p2, p1, sizeof(struct Point2D));
	
	free(p1);
}
```


# 구조체 배열 선언하기
```c
struct Point2D {
	int x;
	int y;
};
struct Point3D {
	int x;
	int y;
};
int main34() {
	struct Point2D p[3];

	p[0].x = 10;
	p[0].y = 20;
	p[1].x = 30;
	p[1].y = 40;
	p[2].x = 50;
	p[2].y = 60;

	printf("%d %d\n", p[0].x, p[0].y);
	printf("%d %d\n", p[1].x, p[1].y);
	printf("%d %d\n", p[2].x, p[2].y);

	//구조체 포인터 배열 선언
	struct Point3D *q[3];

	for (int i = 0; i < sizeof(q) / sizeof(struct Point3D*); i++)
	{
		q[i] = malloc(sizeof(struct Point3D));
	}

	q[0]->x = 10;
	q[0]->y = 20;
	q[1]->x = 30;
	q[1]->y = 40;
	q[2]->x = 50;
	q[2]->y = 60;

	printf("%d %d\n", q[0]->x, q[0]->y);
	printf("%d %d\n", q[1]->x, q[1]->y);
	printf("%d %d\n", q[2]->x, q[2]->y);

	for (int i = 0; i < sizeof(q) / sizeof(struct Point3D*); i++)
	{
		free(q[i]);
	}

	return 0;
}
```

# 공용체 사용하기
```c
union Box {
	short candy;
	float snack;
	char doll[8];
};
union Data {
	char c1;
	short num1;
	int num2;
};
int main35() {
	//구조체는 각 한자리씩 차지하지만 공용체는 가장큰 메모리를 공유한다.

	union Box b1;

	printf("%d\n", sizeof(b1)); // 8

	strcpy(b1.doll, "bear");

	printf("%d\n", b1.candy); // 25954
	printf("%f\n", b1.snack); // 4464428256607..... 
	printf("%s\n", b1.doll); // bear
	//가장큰 곳 에 할당을 해서 한꺼번에 값 을 넣으면 안된다. 따로따로 진행해야함

	b1.candy = 10;
	printf("%d\n", b1.candy);

	b1.snack = 60000.0f;
	printf("%f\n", b1.snack);

	//공용체와 엔디언
	union Data d1;

	d1.num2 = 0x12345678;
	printf("0x%x", d1.num2); //0x12345678
	printf("0x%x", d1.num1); //0x5678
	printf("0x%x", d1.c1); //0x78 1byte 만 나옴(리틀 엔디언) 메모리에 거꾸로 저장되어있음.
	return 0;

	//공용체 포인터 선언하고 메모리 할당
	union Box* t1 = malloc(sizeof(union Box));

	printf("%d\n", sizeof(union Box));

	strcpy(t1->doll, "bear");

	printf("%d\n", t1->candy);

	free(t1);

	return 0;
}
```
